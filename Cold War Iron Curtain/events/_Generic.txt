add_namespace = generic

# Get Annexed or War, 50/50 chance

  id = generic.2
	title = generic.1.t
	desc = generic.1.d

	is_triggered_only = yes

# Get Annexed
	option = {
		name = generic.1.a
		FROM = { annex_country = { target = ROOT } }

		}
	}

# War
	option = {
		name = generic.1.b
		FROM = { declare_war_on = { target = ROOT type = annex_everything } }
	  ai_chance = {
		  factor = 50
		}
	}
}
