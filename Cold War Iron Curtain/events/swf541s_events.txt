	id = swf.1
	hidden = yes

	trigger = {
		tag = USA
		date > 1950.10.31
		date < 1950.11.5
		NOT = { has_global_flag = truman_assasination_attempt }
		NOT = { has_global_flag = truman_assasinated }
		has_capitulated = no
	}

	mean_time_to_happen = { days = 1 }
	fire_only_once = yes


	immediate = {
		random_list = { 
			80 = { news_event = { hours = 6 id = swf.2 } } 
			20 = { news_event = { hours = 6 id = swf.3 } } 
		}	
	}
}
