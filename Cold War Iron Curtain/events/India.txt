add_namespace = india

#Indo-Pakistani War of 9 september 1965 
country_event = {
	id = india.1
	title = india.1.t
	desc = india.1.d
	picture = GFX_report_event_soviet_tanks_snow
	
	is_triggered_only = yes
	
	fire_only_once = yes
	
	option = {
		name = india.1.a
		set_global_flag = indo_pakistani_war_1965
		RAJ = {
			declare_war_on = {
					type = take_state
					target = PAK
					generator = { 779 }
			}
		}
	}
}

#Surrender when India has taken too much land
country_event = {
	id = india.3
	title = india.3.t
	desc = india.3.d
	picture = GFX_report_event_soviet_tanks_snow
	

	is_triggered_only = yes

	fire_only_once = yes

	#Surrender and offer territory
	option = {
		name = india.3.a

		RAJ = {
			country_event = india.4
		}
		custom_effect_tooltip = india.3.a_tt
		effect_tooltip = {
			RAJ = { 
				white_peace = {
					tag = PAK 
					message = PAK_agree_peace 
				} 
			}

		}
	}
}
