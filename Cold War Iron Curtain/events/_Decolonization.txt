#British Cameroons - Mandate Expired
country_event = {
	id = decolonization.2
	title =decolonization.2.t	
	desc = decolonization.2.d
	
	is_triggered_only = yes
	fire_only_once = yes
	
	option = {
		name = decolonization.2.a
		if = {
	        limit = {
		        country_exists = CMR
				country_exists = NIA
	        }
	        CMR = { transfer_state = 1105 }
			NIA = { transfer_state = 1106 }
        }
		if = {
	        limit = {
		        country_exists = NIA
			    NOT = { country_exists = CMR }
	        }
	        NIA = { transfer_state = 1105 transfer_state = 1106 }
        }
	    if = {
	        limit = {
		        country_exists = CMR
			    NOT = { country_exists = NIA }
	        }
	        CMR = { transfer_state = 1105 transfer_state = 1106 }
        }
	}
}
